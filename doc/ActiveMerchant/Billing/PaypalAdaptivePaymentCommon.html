  base.cattr_accessor :test_redirect_url

  base.cattr_accessor :live_redirect_pre_approval_url

  base.live_redirect_pre_approval_url = 'https://www.paypal.com/webscr?cmd=_ap-preapproval&preapprovalkey='
end
